剪枝有：
- 优化搜索顺序：先对数组按照从大到小排序，减少搜索分支
- 可行性剪枝：若当前组内数大于target，返回
- 排除等效冗余：在同一组内，从start开始搜。当开了新的一组后，从0开始搜。这种方法搜出来的方案每组是按照组合而不是排列
- https://www.acwing.com/solution/content/13372/
```
class Solution {
    public boolean canPartitionKSubsets(int[] nums, int k) {
        int sum = 0;
        int max = 0;
        for(int num : nums) {
            sum += num;
            max = num > max ? num : max;
        }
        
        if(sum % k != 0 || max > sum / k)
            return false;
        sum = sum / k;
        
        boolean[] visited = new boolean[nums.length];
        return dfs(0, 0, sum,  k, visited, nums);
    }
    
    private boolean dfs(int index, int curSum, int target, int k, boolean[] visited, int[] nums) {
        if(k == 0)
            return true;
        
        if(curSum == target) {
            return dfs(0, 0, target, k - 1, visited, nums);
        }
        
        for(int i = index; i < nums.length; i ++) {
            if(visited[i] || nums[i] + curSum > target)
                continue;
            
            visited[i] = true;
            if(dfs(i + 1, curSum + nums[i], target, k, visited, nums))
                return true;
            
            visited[i] = false;
        }
        return false;
    }
}
```
